CREATE TABLE public.business_profiles (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL UNIQUE,
  legal_name text,
  trade_name text,
  constitution text,
  business_pan text,
  email text,
  phone_number text,
  gstin text,
  portal_username text,
  portal_password text,
  nature_of_business text,
  state text,
  date_of_registration date,
  filing_frequency text,
  annual_turnover_range text,
  registered_address text,
  is_profile_complete boolean DEFAULT false,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  hsn_codes text,
  sac_codes text,
  CONSTRAINT business_profiles_pkey PRIMARY KEY (id),
  CONSTRAINT business_profiles_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.files (
  file_id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid,
  file_url text NOT NULL,
  status text DEFAULT 'pending'::text,
  file_path text,
  created_at timestamp with time zone DEFAULT now(),
  invoice_number text,
  company_name text,
  CONSTRAINT files_pkey PRIMARY KEY (file_id),
  CONSTRAINT files_user_id_fkey FOREIGN KEY (user_id) REFERENCES public.users(user_id)
);
CREATE TABLE public.gst_invoices (
  gst_invoice_id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid,
  invoice_url text NOT NULL,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT gst_invoices_pkey PRIMARY KEY (gst_invoice_id),
  CONSTRAINT gst_invoices_user_id_fkey FOREIGN KEY (user_id) REFERENCES public.users(user_id)
);
CREATE TABLE public.pdf (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  company_name text,
  invoice_number text,
  status text,
  reason text,
  links text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT pdf_pkey PRIMARY KEY (id)
);
CREATE TABLE public.sale_purchase_register (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL,
  sales_register_url text,
  purchase_url text,
  gstr1b_url text,
  reconciliation_json jsonb,
  gstr9b_url text,
  created_at timestamp with time zone DEFAULT now(),
  gstr2b_url text,
  gstr3b_url text,
  CONSTRAINT sale_purchase_register_pkey PRIMARY KEY (id),
  CONSTRAINT sale_purchase_register_user_id_fkey FOREIGN KEY (user_id) REFERENCES public.users(user_id)
);
CREATE TABLE public.users (
  user_id uuid NOT NULL DEFAULT gen_random_uuid(),
  phone_number text NOT NULL UNIQUE,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT users_pkey PRIMARY KEY (user_id)
